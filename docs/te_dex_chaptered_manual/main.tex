\newcommand{\figureslot}[2]{%
  \begin{figure}[htbp]
    \centering
    \IfFileExists{#1}{%
      \includegraphics[width=0.85\linewidth]{#1}%
    }{%
      \fcolorbox{slotline}{slotbg}{%
        \begin{minipage}[c][1.45in][c]{0.82\linewidth}
          \centering
          \textbf{Missing figure}\\[0.4em]
          \small Expected file:\\
          \nolinkurl{#1}
        \end{minipage}
      }%
    }%
    \caption{#2}
  \end{figure}
}
\newcommand{\figurepanel}[2]{%
  \begin{minipage}[t]{0.48\linewidth}
    \centering
    \IfFileExists{#1}{%
      \includegraphics[width=\linewidth]{#1}%
    }{%
      \fcolorbox{slotline}{slotbg}{%
        \begin{minipage}[c][1.35in][c]{0.92\linewidth}
          \centering
          \textbf{Missing figure}\\[0.35em]
          \small\nolinkurl{#1}
        \end{minipage}
      }%
    }%
    \par\vspace{0.35em}
    {\small #2}
  \end{minipage}%
}
\newcommand{\svgfigurepanel}[2]{%
  \begin{minipage}[t]{0.48\linewidth}
    \centering
    \IfFileExists{#1}{%
      \ifnum\pdfshellescape=1
        \includesvg[width=\linewidth]{#1}%
      \else
        \fcolorbox{slotline}{slotbg}{%
          \begin{minipage}[c][1.35in][c]{0.92\linewidth}
            \centering
            \textbf{SVG requires shell escape}\\[0.35em]
            \small\nolinkurl{#1}
          \end{minipage}
        }%
      \fi
    }{%
      \fcolorbox{slotline}{slotbg}{%
        \begin{minipage}[c][1.35in][c]{0.92\linewidth}
          \centering
          \textbf{Missing SVG}\\[0.35em]
          \small\nolinkurl{#1}
        \end{minipage}
      }%
    }%
    \par\vspace{0.35em}
    {\small #2}
  \end{minipage}%
}
